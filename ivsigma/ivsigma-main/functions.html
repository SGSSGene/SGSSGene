Utility functions

Convert chars to ranks

  1. void ivs::convert_char_to_rank<Alphabet>(std::span<char const> in, std::span<uint8_t> out)
  2. auto ivs::convert_char_to_rank<Alpabet>(std::span<char const> in) -> std::vector<uint8_t>
  3. auto ivs::view_char_to_rank<Alphabet> = /*unspecificed*/

Conversion from char space to rank space. All versions are guaranteed to not throw.

Example

#include <ivsigma/ivsigma.h>
#include <iostream>

int main()
{
    auto input = std::string{"ACCACGT"};
    { // Style 1
        auto output = std::vector<uint8_t>{};
        output.resize(input.size());
        ivs::convert_char_to_rank<ivs::dna4>(input, output);
        for (auto r : output) {
            std::cout << (int)r;
        }
        std::cout << '\n';
    }
    { // Style 2
        std::vector<uint8_t> output = ivs::convert_char_to_rank<ivs::dna4>(input);
        for (auto r : output) {
            std::cout << (int)r;
        }
        std::cout << '\n';
    }
    { // Style 3
        auto output_view = input | ivs::view_char_to_rank<ivs::dna4>;
        auto output = std::vector<uint8_t>(output_view.begin(), output_view.end());
        for (auto r : output) {
            std::cout << (int)r;
        }
        std::cout << '\n';
    }
}
